Define notification type:
```typescript
interface Notification {
```

Hooks:
- `useNotifications(params?: { page?: number, limit?: number })`:
  - queryKey: `['notifications', params]`
  - queryFn: GET `${API_ROUTES.NOTIFICATIONS}?page=${params.page}&limit=${params.limit}`
  - Returns `{ data: Notification[], pagination: {...} }`

- `useUnreadCount()`:
  - queryKey: `['notifications', 'unread-count']`
  - queryFn: GET `${API_ROUTES.NOTIFICATIONS}/unread-count`
  - select: `(data) => data.count`
  - `refetchInterval: 30_000` (poll every 30 seconds per RESEARCH.md recommendation)

- `useMarkRead()`:
  - mutationFn: (id: number) => PATCH `${API_ROUTES.NOTIFICATIONS}/${id}/read`
  - onSuccess: invalidate `['notifications']` and `['notifications', 'unread-count']`

- `useMarkAllRead()`:
  - mutationFn: () => PATCH `${API_ROUTES.NOTIFICATIONS}/read-all`
  - onSuccess: invalidate `['notifications']` and `['notifications', 'unread-count']`
  - Shows toast "All notifications marked as read"

```typescript
function timeAgo(dateString: string): string {
  const seconds = Math.floor((Date.now() - new Date(dateString).getTime()) / 1000);
  if (seconds < 60) return 'just now';
  const minutes = Math.floor(seconds / 60);
  if (minutes < 60) return `${minutes}m ago`;
  const hours = Math.floor(minutes / 60);
  if (hours < 24) return `${hours}h ago`;
  const days = Math.floor(hours / 24);
  return `${days}d ago`;
}
```
